function doPost(e) {
  try {
    var data  = JSON.parse(e.postData.contents);
    var emp   = data.emp;
    var fecha = data.fecha;
    var regs  = data.registros;

    var ss    = SpreadsheetApp.getActiveSpreadsheet();
    var sheet = ss.getSheetByName(emp);
    if (!sheet) return resp({ok:false, msg:'Hoja no encontrada: ' + emp});

    // Borrar filas existentes de esa fecha
    var vals = sheet.getDataRange().getValues();
    for (var i = vals.length - 1; i >= 0; i--) {
      var raw = vals[i][0] ? String(vals[i][0]).trim() : '';
      if (toISO(raw) === fecha) sheet.deleteRow(i + 1);
    }

    // Insertar nuevas filas
    var d = fecha.split('-');
    var dis = d[2] + '/' + d[1] + '/' + d[0];
    regs.forEach(function(r) {
      sheet.appendRow([dis, r.tipo, r.hora, '', '', '']);
    });

    return resp({ok:true});
  } catch(err) {
    return resp({ok:false, msg:err.toString()});
  }
}

function toISO(s) {
  var m = s.match(/^(\d{1,2})\/(\d{1,2})\/(\d{4})$/);
  if (m) return m[3]+'-'+('0'+m[2]).slice(-2)+'-'+('0'+m[1]).slice(-2);
  return s;
}

function resp(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}

function doGet() {
  return ContentService.createTextOutput('OK');
