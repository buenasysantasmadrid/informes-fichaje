function doPost(e) {
  try {
    var data   = JSON.parse(e.postData.contents);
    var action = data.action;

    if (action === 'guardarSolicitud') {
      return guardarSolicitud(data);
    } else if (action === 'aprobarSolicitud') {
      return aprobarSolicitud(data);
    } else if (action === 'rechazarSolicitud') {
      return rechazarSolicitud(data);
    }
    return resp({ok:false, msg:'Accion desconocida'});
  } catch(err) {
    return resp({ok:false, msg:err.toString()});
  }
}

function guardarSolicitud(data) {
  var ss    = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName('SOLICITUDES');
  if (!sheet) {
    sheet = ss.insertSheet('SOLICITUDES');
    sheet.appendRow(['ID','Empleado','Fecha','Tipo','Hora actual','Hora nueva','Motivo','Estado','Fecha solicitud','Email empleado']);
  }
  var id = 'SOL-' + Date.now();
  var now = new Date().toLocaleString('es-ES');
  sheet.appendRow([
    id, data.emp, data.fecha, data.tipo,
    data.horaActual, data.horaNueva, data.motivo,
    'pendiente', now, data.emailEmp || ''
  ]);
  return resp({ok:true, id:id});
}

function aprobarSolicitud(data) {
  var ss     = SpreadsheetApp.getActiveSpreadsheet();
  var solSh  = ss.getSheetByName('SOLICITUDES');
  var empSh  = ss.getSheetByName(data.emp);
  if (!solSh || !empSh) return resp({ok:false, msg:'Hoja no encontrada'});

  // Marcar como aprobada en SOLICITUDES
  var vals = solSh.getDataRange().getValues();
  for (var i = 1; i < vals.length; i++) {
    if (vals[i][0] === data.id) {
      solSh.getRange(i+1, 8).setValue('aprobada');
      break;
    }
  }

  // Actualizar/mover la fila en la hoja del empleado
  var empVals = empSh.getDataRange().getValues();
  var fechaNueva = data.fechaNueva || data.fecha;
  for (var j = 0; j < empVals.length; j++) {
    var rowFecha = toISO(String(empVals[j][0] || '').trim());
    var rowTipo  = String(empVals[j][1] || '').toUpperCase().trim();
    var rowHora  = String(empVals[j][2] || '').trim();
    if (rowFecha === data.fecha && rowTipo === data.tipo && rowHora === data.horaActual) {
      if (fechaNueva !== data.fecha) {
        // Cambio de fecha: borrar fila original e insertar en la nueva fecha
        empSh.deleteRow(j + 1);
        var dParts = fechaNueva.split('-');
        var disFecha = dParts[2]+'/'+dParts[1]+'/'+dParts[0];
        empSh.appendRow([disFecha, data.tipo, data.horaNueva, '', '', '']);
      } else {
        // Solo cambio de hora
        empSh.getRange(j+1, 3).setValue(data.horaNueva);
      }
      break;
    }
  }

  // Notificar por email al empleado
  if (data.emailEmp) {
    MailApp.sendEmail({
      to: data.emailEmp,
      subject: 'Solicitud de cambio aprobada · Fichajes RRHH',
      body: 'Hola ' + data.emp.split(' ')[0] + ',\n\n'
        + 'Tu solicitud de cambio ha sido APROBADA:\n\n'
        + '  Fecha: ' + data.fecha + '\n'
        + '  Tipo: ' + data.tipo + '\n'
        + '  Hora anterior: ' + data.horaActual + '\n'
        + '  Hora nueva: ' + data.horaNueva + '\n\n'
        + 'Un saludo,\nRRHH'
    });
  }

  return resp({ok:true});
}

function rechazarSolicitud(data) {
  var ss    = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName('SOLICITUDES');
  if (!sheet) return resp({ok:false});
  var vals = sheet.getDataRange().getValues();
  for (var i = 1; i < vals.length; i++) {
    if (vals[i][0] === data.id) {
      sheet.getRange(i+1, 8).setValue('rechazada');
      break;
    }
  }
  // Notificar rechazo
  if (data.emailEmp) {
    MailApp.sendEmail({
      to: data.emailEmp,
      subject: 'Solicitud de cambio rechazada · Fichajes RRHH',
      body: 'Hola ' + data.emp.split(' ')[0] + ',\n\n'
        + 'Tu solicitud de cambio para el ' + data.fecha
        + ' (' + data.tipo + ' ' + data.horaActual + ') ha sido rechazada.\n\n'
        + 'Si tienes dudas, contacta con RRHH.\n\nUn saludo'
    });
  }
  return resp({ok:true});
}

function toISO(s) {
  var m = s.match(/^(\d{1,2})\/(\d{1,2})\/(\d{4})$/);
  if (m) return m[3]+'-'+('0'+m[2]).slice(-2)+'-'+('0'+m[1]).slice(-2);
  return s;
}
function resp(obj) {
  return ContentService.createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}
function doGet() { return ContentService.createTextOutput('OK'); }
